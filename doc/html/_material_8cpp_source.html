190 GLuint Material::BuildShader( GLuint _program, GLenum _shader_type )
191 {
192  string source;
193  GLenum type;
194  GLuint shader;
195 
196  switch( _shader_type )
197  {
198  case FRAG_SHADER:
199  source = m_f_source;
200  type = GL_FRAGMENT_SHADER;
201  break;
202  case VERT_SHADER:
203  source = m_v_source;
204  type = GL_VERTEX_SHADER;
205  break;
206  case GEOM_SHADER:
207  source = m_g_source;
208  type = GL_GEOMETRY_SHADER;
209  break;
210  case TC_SHADER:
211  source = m_tc_source;
212  type = GL_TESS_CONTROL_SHADER;
213  break;
214  case TE_SHADER:
215  source = m_te_source;
216  type = GL_TESS_EVALUATION_SHADER;
217  break;
218  default:
219  return 0;
220  }
221  //-- check if shaders are loaded correctly (i.e. if file exist)
222  if( source == "null" )
223  return 0;
224  //-- check OpenGL 4 support (when tessellation shaders present)
225  if( (_shader_type == TC_SHADER || _shader_type == TE_SHADER) && !GLEW_ARB_gpu_shader5)
226  {
227  cerr << "WARNING: OpenGL 4 not supported, can't use tessellation shaders!\n";
228  return 0;
229  }
230 
231  //-- shader version. If OpenGL4 is supported, use 400, else 330
232  string version = "#version 330 compatibility\n";
233  if(GLEW_ARB_gpu_shader5)
234  version = "#version 400 core\n";
235 
236  if( _shader_type == TC_SHADER || _shader_type == TE_SHADER )
237  m_has_tessellation_shader = true;
238 
239  //then load, create, compile and attach tessellation shaders
240  shader = glCreateShader(type);
241 
242  //load and link shader
243  source = version + source;
244  const char *s = source.c_str();
245 
246  glShaderSource(shader, 1, &s, NULL);
247  glCompileShader(shader);
248  glAttachShader(_program,shader);
249 
250  return shader;
251 }
