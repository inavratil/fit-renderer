385 bool Texture::LoadImage(const char* filename)
386 {
387  if(!filename)
388  return false;
389 
390  //Opens image
391  if(!Texture::isILInitialized)
392  {
393  ilInit();
394  Texture::isILInitialized = true;
395  }
396 
397  ILuint id = 0;
398  ilGenImages(1, &id);
399  ilBindImage(id);
400 
401  if(!ilLoadImage(filename))
402  {
403  ShowMessage("Cannot open texture file!");
404  return false;
405  }
406 
407  //Get image attributes
408  m_width = ilGetInteger(IL_IMAGE_WIDTH);
409  m_height = ilGetInteger(IL_IMAGE_HEIGHT);
410  m_bpp = ilGetInteger(IL_IMAGE_BPP);
411 
412  if((m_width <= 0) || (m_height <= 0) )
413  {
414  ShowMessage("Unknown image type!");
415  return false;
416  }
417 
418  //Allocate memory for image
419  unsigned imageSize = (m_bpp * m_width * m_height);
420  m_imageData = new GLubyte[imageSize];
421 
422  if(m_imageData == NULL)
423  {
424  ShowMessage("Can't allocate image data!");
425  return false;
426  }
427 
428  //Copy pixels
429  if(m_bpp==1)
430  ilCopyPixels(0, 0, 0, m_width, m_height, 1, IL_LUMINANCE, IL_UNSIGNED_BYTE, m_imageData);
431  else if(m_bpp==3)
432  ilCopyPixels(0, 0, 0, m_width, m_height, 1, IL_RGB, IL_UNSIGNED_BYTE, m_imageData);
433  else if(m_bpp==4)
434  ilCopyPixels(0, 0, 0, m_width, m_height, 1, IL_RGBA, IL_UNSIGNED_BYTE, m_imageData);
435 
436  //Unbind and free
437  ilBindImage(0);
438  ilDeleteImage(id);
439 
440  cout<<"Image loaded: "<< filename <<"\n";
441  return true;
442 }
