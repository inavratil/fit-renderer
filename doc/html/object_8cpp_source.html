352 void TObject::Draw(bool tessellate)
353 {
354  //don't draw object with draw_object flag set to false
355  if(!m_draw_object)
356  return;
357 #ifdef VERBOSE
358  cout<<"Drawing "<<m_name << endl;
359 #endif
360 
361  if(m_type != INSTANCE)
362  glBindVertexArray(m_vbo.vao);
363 
364  //set patch parameter
365  int patch = m_drawmode;
366  if(tessellate)
367  {
368  if(m_drawmode == GL_TRIANGLE_STRIP)
369  glPatchParameteri(GL_PATCH_VERTICES, 4);
370  else
371  glPatchParameteri(GL_PATCH_VERTICES, 3);
372  patch = GL_PATCHES;
373  }
374 
375  //different drawing mode: simple vertex array or array with element indices
376  if(m_element_indices)
377  {
378  //it's possible to use geometry instancing (if there is more than 1 object instance)
379  if(m_instances > 1)
380  glDrawElementsInstanced(patch, m_vbo.indices, GL_UNSIGNED_INT , 0, m_instances);
381  else
382  glDrawElements(patch, m_vbo.indices, GL_UNSIGNED_INT , 0);
383  }
384  else
385  {
386  if(m_instances > 1)
387  glDrawArraysInstanced(patch, 0, m_vbo.indices * 3, m_instances);
388  else
389  glDrawArrays(patch, 0, m_vbo.indices * 3);
390  }
391 #ifdef VERBOSE
392  cout<<"...done\n";
393 #endif
394 }
