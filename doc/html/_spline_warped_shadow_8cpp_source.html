824 //-- _P je v intervalu [0..grid_res-1]
825 glm::vec2 SplineWarpedShadow::ComputeDiff( glm::vec2 _P )
826 {
827  glm::vec2 diff( 0.0 );
828 
829  if( m_iTexID == 0 || m_pFuncValues == NULL || !m_bEnabled ) return diff;
830 
831  //-- vypocet souradnice bunky, ve ktere se bod _P nachazi
832  glm::vec2 grid_coords = glm::floor( _P );
833 
834  glm::mat4 P;
835  glm::vec4 Tx, Ty;
836  glm::vec4 Q;
837 
838  Splines4x4 splines;
839 
840  //-- prevod do intervalu [0..1], tj. pozice v bunce
841  float x = glm::fract(_P.x);
842  float y = glm::fract(_P.y);
843 
844  splines = _GetSplinesValues( grid_coords );
845 
846  Tx = glm::vec4( 1.0, x, glm::pow(x, 2.0f), glm::pow(x,3.0f) );
847  Ty = glm::vec4( 1.0, y, glm::pow(y, 2.0f), glm::pow(y, 3.0f) );
848 
849  //-- diff X
850 
851  //-- definice po sloupcich
852  P = glm::mat4(
853  splines.p0[0].x, splines.p1[0].x, splines.p2[0].x, splines.p3[0].x,
854  splines.p0[1].x, splines.p1[1].x, splines.p2[1].x, splines.p3[1].x,
855  splines.p0[2].x, splines.p1[2].x, splines.p2[2].x, splines.p3[2].x,
856  splines.p0[3].x, splines.p1[3].x, splines.p2[3].x, splines.p3[3].x
857  );
858 
859  Q = P * Mcr * Tx;
860  diff.x = glm::dot(Q * Mcr, Ty);
861 
862  //-- diff Y
863 
864  //-- definice po sloupcich
865  P = glm::mat4(
866  splines.p0[0].y, splines.p1[0].y, splines.p2[0].y, splines.p3[0].y,
867  splines.p0[1].y, splines.p1[1].y, splines.p2[1].y, splines.p3[1].y,
868  splines.p0[2].y, splines.p1[2].y, splines.p2[2].y, splines.p3[2].y,
869  splines.p0[3].y, splines.p1[3].y, splines.p2[3].y, splines.p3[3].y
870  );
871 
872  Q = P * Mcr * Tx;
873  glm::vec4 v = Mcr * Ty;
874  diff.y = glm::dot(Q * Mcr, Ty);
875 
876  return diff * POLY_BIAS;
877 }

880 Splines4x4 SplineWarpedShadow::_GetSplinesValues( glm::vec2 _c )
881 {

893 
894  //-- jedna bunka
895  // P3 ----
896  // | |
897  // P2 ----
898  // |XX|
899  // P1 ----
