18 bool TScene::CreateShadowMap(vector<TLight*>::iterator ii)

24  if((*ii)->GetType() == SPOT)

46  //glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_COMPARE_MODE, GL_COMPARE_REF_TO_TEXTURE);
47  //glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_COMPARE_FUNC, GL_LEQUAL);
48  }
49 
51  glGenFramebuffers(1, &(*ii)->fbo);
52  glBindFramebuffer(GL_FRAMEBUFFER, (*ii)->GetFBO());
53  //add shadow texture(s)
54  if((*ii)->GetType() == OMNI)
55  glFramebufferTextureLayer(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, *(*ii)->GetShadowTexID(), 0, 0); //attach first texture layer
56  else
57  glFramebufferTexture(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, *(*ii)->GetShadowTexID(), 0);
58  glDrawBuffer(GL_NONE);
59  glReadBuffer(GL_NONE);
60  //check FBO creation
61  if(!CheckFBO())
62  {
63  ShowMessage("ERROR: FBO creation for shadow map failed!",false);
64  return false;
65  }
66 
67  return true;
68 }
